## 폰켓몬

```java
import java.util.ArrayList;
import java.util.List;

class Solution {
    public int solution(int[] nums) {
        int answer = 0;
        int length = nums.length / 2;
        List<Integer> list = new ArrayList<>();

        for (int i = 0; i < nums.length; i++) {
            if (!list.contains(nums[i])) {
                list.add(nums[i]);
            }

            if (list.size() == length) {
                return list.size();
            }
            
            if (i == nums.length - 1) {
                return list.size();
            }
        }

        return answer;
    }
}
```
